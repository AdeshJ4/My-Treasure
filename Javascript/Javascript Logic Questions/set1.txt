function countVowels (str){
  let vowels = 'aeiou';
  let count = 0;
  for (let ch of str){
    if(vowels.includes(ch.toLowerCase())){
      count++;
    }
  }

}

function palindrome(str) {
  let reverseStr = "";
  for (let i = str.length - 1; i >= 0; i--) {
    reverseStr += str[i];
  }

  if (reverseStr === str) {
    return true;
  }
  return false;
}

function alphabeticalOrder(str){
    for(let i=0; i<str.length; i++){
        let isSorted = false;
        for(let j=1; j<str.length-i; j++){
            if(str[j] < str[j-1]){ 
                [str[j], str[j-1]] = [str[j-1],str[j]]
                isSorted = true;
            }
        }
        if(!isSorted){

        }
    }
}

function mostFrequentElementOfArray(array) {
  const obj = {};
  for (let num of array) {
    if (obj.hasOwnProperty(num)) {  // if(obj[num]){...rest code}
      obj[num]++;
    } else {
      obj[num] = 1;
    }
  }
  let maxValue = 0;
  let maxProperty = [];
  for (let key in obj) {    // 1>2, 2>2
    if (obj[key] > maxValue) {
      maxValue = obj[key];
      maxProperty = [key]; // p
    } else if (obj[key] === maxValue) {
      maxProperty.push(key); // e  -> p, e
    }
  }
  return maxProperty;
}

Question 21 : Shuffle array
INPUT: [10, 20, 30, 40]
OUTPUT: [40, 10, 30, 20]


function shuffleArray(array){
  for(let i = array.length-1; i>0; i--){
    const j = Math.floor(Math.random() * (i+1));  // 0 - current index
    [array[i], array[j]] = [array[j], array[i]]
  }
}

const arr = [10, 20, 30, 40];
console.log('Before: ', arr);
shuffleArray(arr)
console.log('After ', arr);

      let isDuplicate = false; // false, false, false
