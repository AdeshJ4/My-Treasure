Array: 
=====
const arr = [10, 20, 30, 40];

for (let num of arr){
  console.log(num);
}

for (let key in arr){
  console.log(key, arr[key]);   // key -> index, value-> arr[key]
}




object: 
=======
const obj = {
  fname: "Adesh",
  lname: "Jadhav",
  age: 22,
};

for (let key in obj){
  console.log(key, obj[key]);  // key -> properties, values -> obj[key]
}

for (let key of Object.keys(obj)){
  console.log(key, obj[key])
}
