Daisy UI:

-> component library for Tailwind CSS
 
https://daisyui.com/

step 1: install Package: npm i -D daisyui@latest
step 2: Add daisyUI to tailwind.config.js:

const config: Config = {
  plugins: [require('daisyui')],   <-------- Add Plugins
  daisyui: {
    themes: ["forest", "winter", "retro"],   <-------- Thems
  },
};
export default config;
