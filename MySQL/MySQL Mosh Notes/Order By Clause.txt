Order By 

1] Alphabetically [A - Z]
-> 
select *
from customers 
order by col_name;

2] Alphabetically [Z - A]
-> 
select *
from customers 
order by col_name DESC; 

3] Sort By col1(state) and if there is double entry then sort that by second col(first_name)
-> 
select *
from customers 
order by state, first_name;

select *
from customers 
order by state desc, first_name desc;

select *
from customers 
order by state desc, first_name;



// Here first_name = 1, last_name = 2
-> 
select first_name, last_name
from customers
order by 1, 2;
