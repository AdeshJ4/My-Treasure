													***Filter data***
-------------------------------------------------------------------------------------------------
-------------------------------------------------------------------------------------------------
***Greater than (>)***
select *
from customers
where points > 3000;
-------------------------------------------------------------------------------------------------
***Less than (<)***
select *
from customers
where points < 3000;
-------------------------------------------------------------------------------------------------
***Peoples from 'America'***
select *
from customers
where state = 'America';
-------------------------------------------------------------------------------------------------
***Peoples not From 'America'***
select *
from customers
where state != 'America';
-------------------------------------------------------------------------------------------------
***Select birth_Date which is greater than 'year-month-date' ***
select *
from customers
where birth_date > '1992-01-01';
-------------------------------------------------------------------------------------------------
****Get Orders date of only this year***
select *
from orders
where order_date >= '2018-01-01' AND order_date<= '2018-12-30';
